The marker is at {{marker.position}}

JavaScript 

$scope.map = {
  center: [47.5, -122.5],
  options: function() {
    return {
      streetViewControl: false,
      scrollwheel: false
    }
  }
};

$scope.click = function() {
  alert($scope.marker.position);
}

$scope.marker = {
  position: [47.5, -122.5],
  decimals: 4,
  options: function() {
    return { draggable: true };
  }
}

HTML 

<map ng-transclude class="google-map" center="map.center" options="map.options"> 

  <marker position="marker.position" decimals="marker.decimals" options="marker.options"></marker>

  <control position="rightBottom">
    <div class="control-example">
      <h5>The marker is at {{marker.position}}</h5>
      <button class="btn btn-primary" ng-click='click()'>Click Me</button>
    </div>
  </control>

</map>
