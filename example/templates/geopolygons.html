Creates polygons from a GeoJSON Feature Collection of polygons. Requires a URL to a .GeoJSON file. This example includes how to style map features.

HTML 

<div map ng-transclude class="google-map" center="map.center" options="map.options"> 
	<geopoints url="states.url" options="states.options" events="states.events"></geopoints>
</div>

JavaScript 

$scope.states = {
	url: "../data/AirNow_Sites_PM2.5.geojson",
	events: {
		click: function(e, m) {
			var lat = e.latLng.lat();
			var lng = e.latLng.lng();
			var name = m.getProperty("siteName");
			alert(lat + " " + lng + " " + name);
		}
	}
}
