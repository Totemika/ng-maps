Creates multiple markers from a GeoJSON Feature Collection of points. Requires a URL to a .GeoJSON file. This example includes how to bind events to markers.

HTML 

<div map ng-transclude class="google-map" center="map.center" options="map.options"> 
	<geopoints url="stations.url" options="stations.options" events="stations.events"></geopoints>
</div>

JavaScript 

$scope.stations = {
	url: "../data/AirNow_Sites_PM2.5.geojson",
	events: {
		click: function(e, m) {
			var lat = e.latLng.lat();
			var lng = e.latLng.lng();
			var name = m.getProperty("siteName");
			alert(lat + " " + lng + " " + name);
		}
	}
}
